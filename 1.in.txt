void qsort(int levi,int desni)
{
     if (levi < desni)
     {
       int i=levi;
       int j=desni;
       double tmp;
       double piv1=slope[(levi+desni)/2];
       while (i<=j)
       {
         while (niz[i]<piv1) i++;
         while (niz[j]>piv1) j--;
         if (i<=j)
         {
           swap(niz[i], niz[j]);
           i++;
           j--;
         }
       }
       qsort(levi,j);
       qsort(i,desni);
     }
}
